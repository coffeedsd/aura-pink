// ── CHATS ADMIN ────────────────────────────────────────── let currentSessionId = null; let chatPollInterval = null; let lastAdminMsgTime = null; const SB_CHAT = 'https://acylsytogrgyesrcbthx.supabase.co'; const SB_CHAT_KEY = 'sb_publishable_sTp-NwKK2dqaMjjMDcqDsA_HSl45kVD'; const chatHeaders = { 'apikey': SB_CHAT_KEY, 'Authorization': `Bearer ${SB_CHAT_KEY}`, 'Content-Type': 'application/json', 'Prefer': 'return=representation' }; async function loadChats() { try { const r = await fetch(`${SB_CHAT}/rest/v1/chat_sessions?order=updated_at.desc`, { headers: chatHeaders }); const sessions = await r.json(); const list = document.getElementById('chat-sessions-list'); if (!list) return; const openCount = sessions.filter(s => s.status === 'open').length; const badge = document.getElementById('chats-badge'); if (badge) { badge.textContent = openCount; badge.style.display = openCount > 0 ? 'inline' : 'none'; } list.innerHTML = sessions.length === 0 ? '

Нет обращений

' : ''; sessions.forEach((s, i) => { const clientNum = s.client_id.replace('client_', '#').slice(0, 12); const div = document.createElement('div'); div.style.cssText = `padding:0.85rem 1rem;cursor:pointer;border-bottom:1px solid var(--color-border,#e8e0d8);transition:background 0.15s;${currentSessionId === s.id ? 'background:#f5f0eb;' : ''}`; div.innerHTML = `
Клиент ${clientNum} ${s.status === 'open' ? 'Открыт' : 'Закрыт'}

${s.last_message || 'Нет сообщений'}

`; div.onmouseenter = () => { if (currentSessionId !== s.id) div.style.background = '#faf8f6'; }; div.onmouseleave = () => { if (currentSessionId !== s.id) div.style.background = ''; }; div.onclick = () => openChat(s); list.appendChild(div); }); } catch(e) { console.error('loadChats error', e); } } async function openChat(session) { currentSessionId = session.id; lastAdminMsgTime = null; if (chatPollInterval) clearInterval(chatPollInterval); const title = document.getElementById('chat-header-title'); const closeBtn = document.getElementById('close-chat-btn'); const replyArea = document.getElementById('admin-reply-area'); const clientNum = session.client_id.replace('client_', '#').slice(0, 12); if (title) title.textContent = `Клиент ${clientNum} — ${session.status === 'open' ? 'открыт' : 'закрыт'}`; if (closeBtn) closeBtn.style.display = session.status === 'open' ? 'inline-block' : 'none'; if (replyArea) replyArea.style.display = session.status === 'open' ? 'flex' : 'none'; await renderChatMessages(session.id); loadChats(); // Polling новых сообщений chatPollInterval = setInterval(() => renderNewMessages(session.id), 3000); } async function renderChatMessages(sid) { const box = document.getElementById('admin-messages'); if (!box) return; const r = await fetch(`${SB_CHAT}/rest/v1/chat_messages?session_id=eq.${sid}&order=created_at.asc`, { headers: chatHeaders }); const msgs = await r.json(); box.innerHTML = ''; msgs.forEach(m => appendAdminMsg(m)); if (msgs.length > 0) lastAdminMsgTime = msgs[msgs.length - 1].created_at; box.scrollTop = box.scrollHeight; } async function renderNewMessages(sid) { if (!lastAdminMsgTime) return; const r = await fetch(`${SB_CHAT}/rest/v1/chat_messages?session_id=eq.${sid}&created_at=gt.${encodeURIComponent(lastAdminMsgTime)}&order=created_at.asc`, { headers: chatHeaders }); const msgs = await r.json(); const box = document.getElementById('admin-messages'); msgs.forEach(m => { appendAdminMsg(m); lastAdminMsgTime = m.created_at; }); if (msgs.length > 0 && box) box.scrollTop = box.scrollHeight; } function appendAdminMsg(m) { const box = document.getElementById('admin-messages'); if (!box) return; const isClient = m.sender === 'client'; const div = document.createElement('div'); div.style.cssText = `display:flex;justify-content:${isClient ? 'flex-start' : 'flex-end'};`; const bubble = document.createElement('div'); bubble.style.cssText = `max-width:70%;padding:0.6rem 0.9rem;border-radius:${isClient ? '4px 12px 12px 12px' : '12px 4px 12px 12px'};background:${isClient ? '#f5f0eb' : 'var(--ink,#2c2420)'};color:${isClient ? 'var(--ink,#2c2420)' : '#fff'};font-size:0.875rem;line-height:1.4;`; bubble.textContent = m.text; div.appendChild(bubble); box.appendChild(div); } async function sendAdminReply() { const input = document.getElementById('admin-reply-input'); const text = input.value.trim(); if (!text || !currentSessionId) return; input.value = ''; try { await fetch(`${SB_CHAT}/rest/v1/chat_messages`, { method: 'POST', headers: chatHeaders, body: JSON.stringify({ session_id: currentSessionId, sender: 'admin', text }) }); await fetch(`${SB_CHAT}/rest/v1/chat_sessions?id=eq.${currentSessionId}`, { method: 'PATCH', headers: chatHeaders, body: JSON.stringify({ last_message: '✓ ' + text, updated_at: new Date().toISOString() }) }); await renderNewMessages(currentSessionId); } catch(e) { console.error('Reply error', e); } } document.addEventListener('keydown', e => { if (e.key === 'Enter' && document.activeElement?.id === 'admin-reply-input') sendAdminReply(); }); async function closeCurrentChat() { if (!currentSessionId) return; await fetch(`${SB_CHAT}/rest/v1/chat_sessions?id=eq.${currentSessionId}`, { method: 'PATCH', headers: chatHeaders, body: JSON.stringify({ status: 'closed' }) }); document.getElementById('close-chat-btn').style.display = 'none'; document.getElementById('admin-reply-area').style.display = 'none'; document.getElementById('chat-header-title').textContent += ' — закрыт'; loadChats(); }
